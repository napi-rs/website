export const getStaticProps = async () => {
  const sponsors = await fetch(`https://sponsors.napi.rs/sponsor.json`).then(
    (res) => res.json(),
  )
  const washedSponsors = {}
  Object.keys(sponsors).map(key => {
    const sponsorSubList = sponsors[key]
    washedSponsors[key] = sponsorSubList.map(item => {
      return {
        name: item.name,
        img: item.avatarUrl,
        url: item.websiteUrl
      }
    })
  })

  return {
    props: {
      ssg: washedSponsors
    }
  }
}
